Module Tyre.Infix

module Infix: sig .. end
val (<|>) : 'a Tyre.t -> 'b Tyre.t -> [ `Left of 'a | `Right of 'b ] Tyre.t
t <|> t' is alt t t'.
val (<&>) : 'a Tyre.t -> 'b Tyre.t -> ('a * 'b) Tyre.t
t <&> t' is seq t t'.
val ( *> ) : 'b Tyre.t -> 'a Tyre.t -> 'a Tyre.t
ti *> t is prefix ti t.
val ( <* ) : 'a Tyre.t -> 'b Tyre.t -> 'a Tyre.t
t <* ti is suffix t ti.
